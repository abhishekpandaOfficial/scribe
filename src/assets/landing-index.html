③ Knowledge that compounds. The longer you use Scribe, the smarter your workspace becomes

weekly-digest.ts● ready
import { Scribe } from "@scribe/sdk";

// init with a workspace-scoped token
const scribe = new Scribe({ token: process.env.SCRIBE_TOKEN });

// schedule a weekly AI-generated digest
export const digest = scribe.workflows.schedule({
  name: "Board Digest",
  cron: "0 18 * * SUN",
  async run({ ai, docs, slack }) {
    const highlights = await ai.summarize({
      query: "wins, risks and decisions, last 7 days",
      sources: await docs.recent({ days: 7 }),
      cite: true,
    });

    const page = await docs.create({
      template: "board-update",
      data: { highlights },
    });

    await slack.post("#leadership", page.url);
  },
});
